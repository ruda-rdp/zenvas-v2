## Zenvas Data Model Extension

```typescript
// New: Content Series (replaces Order for this type)
interface ContentSeries {
  id: string;
  brandId: string;
  
  // Series info
  title: string;                   // "Kerajaan Nusantara"
  session: string;                 // "Session 1"
  description: string;
  totalEpisodes: number;           // 9
  
  // Content specs
  format: 'VERTICAL' | 'HORIZONTAL' | 'SQUARE';
  durationTarget: number;          // Target seconds per episode (180)
  platforms: Platform[];            // ['YOUTUBE_SHORTS', 'INSTAGRAM_REELS', 'FACEBOOK_REELS']
  
  // Timeline
  startDate: Date;
  targetEndDate: Date;
  targetPublishDate: Date;         // When session should be complete
  
  // Production approach
  aiPipeline: AIPipelineConfig;
  
  // Episodes
  episodes: Episode[];
  
  // Metrics
  totalViews?: number;
  totalSubscribersGained?: number;
  
  // Metadata
  createdAt: DateTime;
  updatedAt: DateTime;
}

interface Episode {
  id: string;
  seriesId: string;
  episodeNumber: number;           // 1-9
  
  // Content
  title: string;
  synopsis: string;
  script: Script;
  
  // Characters in this episode
  characterScenes: CharacterScene[];
  
  // Production status
  status: EpisodeStatus;
  
  // Timeline
  startDate?: Date;
  targetPublishDate?: Date;
  actualPublishDate?: Date;
  
  // Performance (after publish)
  views?: number;
  likes?: number;
  comments?: number;
  watchTime?: number;              // Minutes
  retention?: number;               // Percentage
  
  // AI Assets used
  assets: AIAsset[];
}

enum EpisodeStatus {
  PLANNING = 'planning',           // Script not written
  SCRIPTING = 'scripting',        // Writing episode script
  ASSET_CREATION = 'asset_creation', // Generating AI assets
  PRODUCTION = 'production',       // Video generation
  POST_PRODUCTION = 'post_production', // Editing
  REVIEW = 'review',              // Quality check
  PUBLISHED = 'published',
  FAILED = 'failed'
}

interface AIPipelineConfig {
  // Which AI tools for each stage
  characterGeneration: string;      // 'midjourney' | 'flux' | 'stable_diffusion'
  voiceCloning: string;            // 'elevenlabs' | 'xtts'
  videoGeneration: string;          // 'runway' | 'pika' | 'kling' | 'sora'
  
  // Cost tracking
  costPerEpisode?: number;         // Estimated AI cost
  actualCostPerEpisode?: number;   // Real AI cost
  
  // Quality settings
  qualityPreset: 'FAST' | 'BALANCED' | 'HIGH_QUALITY';
}

interface AIAsset {
  id: string;
  episodeId: string;
  type: AssetType;
  
  // Asset details
  description: string;
  prompt?: string;
  aiModel?: string;
  aiSettings?: object;
  
  // File reference
  fileUrl: string;                 // URL to generated asset
  thumbnailUrl?: string;
  
  // Metadata
  generationTime: number;          // Seconds to generate
  cost: number;                   // Cost in cents/USD
  qualityRating?: number;          // Human rating 1-5
  status: 'PENDING' | 'GENERATING' | 'READY' | 'REJECTED' | 'USED';
  
  createdAt: DateTime;
}

enum AssetType {
  CHARACTER_REF = 'character_ref',       // Reference image
  CHARACTER_LOOK = 'character_look',     // Consistent look trained
  BACKGROUND = 'background',             // Scene backgrounds
  EXPRESSION = 'expression',             // Character expression sheet
  VOICE_CLONE = 'voice_clone',          // Voice profile
  VIDEO_CLIP = 'video_clip',            // Generated video segment
  SOUND_EFFECT = 'sound_effect',        // SFX
  MUSIC = 'music',                       // Background music
  THUMBNAIL = 'thumbnail'                // Episode thumbnail
}
```
