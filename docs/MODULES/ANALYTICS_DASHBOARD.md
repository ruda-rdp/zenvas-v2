# Purpose

Analytics Dashboard provides business intelligence for creative studios: revenue analytics, project performance, team productivity, and client insights.

---

# Data Model

```prisma
// Analytics Report
model AnalyticsReport {
  id              String   @id @default(cuid())
  
  name            String
  type            ReportType
  
  // Date range
  startDate       DateTime
  endDate         DateTime
  
  // Data (cached)
  data            Json     // Pre-computed analytics data
  
  // Generated by
  generatedBy    String
  generatedAt    DateTime @default(now())
  
  createdAt       DateTime @default(now())
}

enum ReportType {
  REVENUE
  PROJECT_PERFORMANCE
  TEAM_PRODUCTIVITY
  CLIENT_ANALYTICS
  LEAD_FUNNEL
  CUSTOM
}

// Analytics Metric
model AnalyticsMetric {
  id              String   @id @default(cuid())
  
  name            String   // e.g., "Total Revenue"
  category        String   // e.g., "Revenue", "Projects"
  
  // Value (most recent)
  value           Float
  previousValue   Float?   // For comparison
  unit            String?   // e.g., "USD", "count", "%"
  
  // Change
  changePercent   Float?
  changeDirection String?  // "up", "down", "stable"
  
  // Period
  period          String   // e.g., "This Month", "Last 30 Days"
  
  updatedAt       DateTime @updatedAt
}

// Dashboard Widget
model DashboardWidget {
  id              String   @id @default(cuid())
  
  // Position
  userId          String?
  
  type            WidgetType
  title           String
  
  // Config
  config          Json     // Widget-specific configuration
  
  // Position
  positionX       Int      @default(0)
  positionY       Int      @default(0)
  width           Int      @default(1)
  height          Int      @default(1)
  
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt
}

enum WidgetType {
  KPI_CARD        // Single metric
  LINE_CHART      // Time series
  BAR_CHART       // Comparison
  PIE_CHART       // Distribution
  TABLE           // Data table
  FUNNEL          // Conversion funnel
  GAUGE           // Progress/score
}
```

# API Contracts

### GET `/api/analytics/dashboard`
Get dashboard metrics.

```typescript
// Response 200 OK
{
  "metrics": [
    {
      "id": "m_001",
      "name": "Total Revenue",
      "category": "Revenue",
      "value": 125000,
      "previousValue": 100000,
      "unit": "USD",
      "changePercent": 25,
      "changeDirection": "up",
      "period": "This Month"
    }
  ],
  "charts": [
    {
      "type": "LINE_CHART",
      "title": "Revenue Over Time",
      "data": {
        "labels": ["Jan", "Feb", "Mar", "Apr", "May", "Jun"],
        "datasets": [
          { "label": "Revenue", "data": [20, 25, 22, 30, 28, 35] }
        ]
      }
    }
  ]
}
```

### GET `/api/analytics/revenue`
Get detailed revenue analytics.

```typescript
// Request
GET /api/analytics/revenue?startDate=2026-01-01&endDate=2026-06-30&groupBy=month

// Response 200 OK
{
  "summary": {
    "totalRevenue": 750000,
    "averageProjectValue": 25000,
    "projectsCompleted": 30

  ],
  "byMonth": [
    { "month": "Jan", "revenue": 100000 },
    { "month": "Feb", "revenue": 125000 }
  ]
}
```
