# Purpose

AI Summary uses artificial intelligence to automatically generate meeting summaries, extract action items, and create recaps from discussions and video calls.

---

# Data Model

```prisma
// AI Summary
model AISummary {
  id              String   @id @default(cuid())
  
  // Source
  sourceType      SummarySourceType
  sourceId        String   // Discussion ID, Call Recording ID, etc.
  
  // Content
  title          String
  
  // Summary
  summary        String   // Main summary
  keyPoints      String[] // Key takeaways
  actionItems    ActionItem[]
  
  // AI info
  aiModel       String?  // GPT-4, Claude, etc.
  confidence     Float?   // AI confidence score
  
  // Status
  status        SummaryStatus @default(PENDING)
  
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt
}

enum SummarySourceType {
  DISCUSSION
  VIDEO_CALL
  TRANSCRIPTION
  DOCUMENT
  EMAIL
}

enum SummaryStatus {
  PENDING
  PROCESSING
  COMPLETED
  FAILED
}

// Action Item
model ActionItem {
  id              String   @id @default(cuid())
  
  summaryId      String
  summary         AISummary @relation(fields: [summaryId], references: [id])
  
  // Content
  description     String
  
  // Assignment
  assignedTo     String?
  dueDate        DateTime?
  
  // Status
  status         ActionItemStatus @default(OPEN)
  completedAt    DateTime?
  
  createdAt       DateTime @default(now())
}

enum ActionItemStatus {
  OPEN
  IN_PROGRESS
  COMPLETED
  CANCELLED
}

// AI Template
model AISummaryTemplate {
  id              String   @id @default(cuid())
  
  name            String
  description     String?
  
  // Prompt template
  promptTemplate  String
  
  // Settings
  model           String   @default("gpt-4")
  temperature     Float    @default(0.7)
  maxTokens       Int      @default(1000)
  
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt
}
```

# API Contracts

### POST `/api/ai/summarize`
Generate summary from content.

```typescript
// Request
{
  "sourceType": "DISCUSSION",
  "sourceId": "disc_001",
  "template": "meeting-summary",
  "options": {
    "extractActionItems": true,
    "sentiment": true
  }
}

// Response 200 OK
{
  "id": "sum_001",
  "title": "Weekly Team Standup - Jul 21",
  "status": "PROCESSING",
  "estimatedTime": 30
}

// Poll for result
GET /api/ai/summaries/sum_001

// Response (when complete)
{
  "id": "sum_001",
  "title": "Weekly Team Standup - Jul 21",
  "summary": "The team discussed progress on Episode 3...",
  "keyPoints": [
    "Scene 5A is complete",
    "VFX vendor approved the explosion shot",
    "Next review on Friday"
  ],
  "actionItems": [
    {
      "description": "Review final VFX",
      "assignedTo": "Director",
      "dueDate": "2026-07-23"
    }
  ]
}
```

### GET `/api/ai/summaries`
List summaries.

```typescript
// Response 200 OK
{
  "summaries": [
    {
      "id": "sum_001",
      "title": "Weekly Team Standup",
      "sourceType": "DISCUSSION",
      "createdAt": "2026-07-21T10:30:00Z",
      "actionItemCount": 3
    }
  ]
}
```
